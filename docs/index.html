GTest

GTest is a simple test framework for C++.

Example of usage:

example_test.cpp:

#include "testframework.hpp"
namespace test {
GTEST(ExampleTest) {
constexpr auto i1{2};
constexpr auto i2{3};
auto addition = i1 + i2;
auto expectedResult{5};
GCHECK(addition, expectedResult);
}
} // namespace test
